**二叉树**

*二叉树的定义*

~~~cpp
struct BiNode
{
    ElemType data;
    BiNode *lchild, *rchild;
} *BiTree;
~~~

*二叉树的先序遍历 DLR*

~~~cpp
bool PreOrderTraverse(BiTree &T)
{
    if(T == nullptr)
    {
        return true;
    }
    //访问根结点
    visit(T);
    //递归遍历左孩子
    PreOrderTraverse(T->lchild);
    //递归遍历右孩子
    PreOrderTraverse(T->rchild);
}
~~~
